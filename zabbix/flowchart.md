            new_host
            new_link_template_host

                ha_og[(default graphs)]

                end

                    trigger_eth0
                    trigger_sda
                end

                    graph_sda
                end

                end
                h_d_f --"map"-->ger_items
                h_d_f --"map"-->ger_trigger
                h_d_f --"map"-->ger_graph
            end
        end
        zabbix_sender --"demo:eth0,sda,xxx"-->trapper_discovery
        zabbix_agent2 --"get demo data"-->zabbix_sender
    end
